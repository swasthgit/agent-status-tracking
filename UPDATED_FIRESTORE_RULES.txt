rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Admin collection - managers only
    match /admin/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;

      // Admin call logs
      match /callLogs/{logId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }

    // mswasth collection - for TLs and their team agents
    match /mswasth/{userId} {
      // All authenticated users can read any document
      allow read: if request.auth != null;

      // Users can write their own document
      allow write: if request.auth != null && request.auth.uid == userId;

      // Managers can write any document
      allow write: if request.auth != null &&
                      exists(/databases/$(database)/documents/admin/$(request.auth.uid));

      // CallLogs subcollection
      match /callLogs/{logId} {
        allow read, write: if request.auth != null;
      }

      // Assigned calls
      match /assignedCalls/{docId} {
        allow read, write: if request.auth != null;
      }

      // ExoCallDetails
      match /exoCallDetails/{docId} {
        allow read, write: if request.auth != null;
      }
    }

    // Agent collections (agent1, agent2, ..., agent31)
    match /{agentCollection}/{userId} {
      // Allow read for:
      // 1. The agent themselves (request.auth.uid == userId)
      // 2. Any Team Leader (exists in mswasth collection with role teamlead)
      // 3. Any Manager (exists in admin collection)
      allow read: if request.auth != null &&
                     agentCollection.matches('^agent[0-9]+$') &&
                     (request.auth.uid == userId ||
                      exists(/databases/$(database)/documents/mswasth/$(request.auth.uid)) ||
                      exists(/databases/$(database)/documents/admin/$(request.auth.uid)));

      // Allow write only for the agent themselves
      allow write: if request.auth != null &&
                      agentCollection.matches('^agent[0-9]+$') &&
                      request.auth.uid == userId;

      // Agent call logs - TLs and managers can read, only agent can write
      match /callLogs/{logId} {
        allow read: if request.auth != null &&
                       agentCollection.matches('^agent[0-9]+$');
        allow write: if request.auth != null &&
                        agentCollection.matches('^agent[0-9]+$') &&
                        request.auth.uid == userId;
      }

      // Assigned calls - TLs and managers can read, only managers can write
      match /assignedCalls/{docId} {
        allow read: if request.auth != null &&
                    agentCollection.matches('^agent[0-9]+$');
        allow write: if request.auth != null &&
                     exists(/databases/$(database)/documents/admin/$(request.auth.uid));
      }

      // ExoCallDetails - all authenticated users can read/write
      match /exoCallDetails/{docId} {
        allow read, write: if request.auth != null &&
                           agentCollection.matches('^agent[0-9]+$');
      }
    }

    // Partners collection - all authenticated users can read
    match /partners/{partnerId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }

    // Manager can read ALL documents (fallback rule)
    match /{document=**} {
      allow read: if request.auth != null &&
                  exists(/databases/$(database)/documents/admin/$(request.auth.uid));
    }
  }
}
